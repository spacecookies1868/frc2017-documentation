161  void control() {
162  /* Get the motion profile status every loop */
163  talon_.GetMotionProfileStatus(status_);

165  /*
166  * track time, this is rudimentary but that's okay, we just want to make
167  * sure things never get stuck.
168  */
169  if (loopTimeout_ < 0) {
170  /* do nothing, timeout is disabled */
171  } else {
172  /* our timeout is nonzero */
173  if (loopTimeout_ == 0) {
174  /*
175  * something is wrong. Talon is not present, unplugged, breaker
176  * tripped
177  */
178  instrumentation::OnNoProgress();
179  } else {
180  --loopTimeout_;
181  }
182  }
183 
184  /* first check if we are in MP mode */
185  if (talon_.GetControlMode() != CANSpeedController::kMotionProfile){
186  /*
187  * we are not in MP mode. We are probably driving the robot around
188  * using gamepads or some other mode.
189  */
190  state_ = 0;
191  } else {
192  /*
193  * we are in MP control mode. That means: starting Mps, checking Mp
194  * progress, and possibly interrupting MPs if thats what you want to
195  * do.
196  */
197  switch (state_) {
198  case 0: /* wait for application to tell us to start an MP */
199  if (bStart_) {
200  bStart_ = false;
201 
202  setValue_ = CANTalon::SetValueMotionProfileDisable;
203  startFilling();
204  /*
205  * MP is being sent to CAN bus, wait a small amount of time
206  */
207  state_ = 1;
208  loopTimeout_ = kNumLoopsTimeout;
209  }
210  break;
211  case 1: /*
212  * wait for MP to stream to Talon, really just the first few
213  * points
214  */
215  /* do we have a minimum numberof points in Talon */
216  if (status_.btmBufferCnt > kMinPointsInTalon) {
217  /* start (once) the motion profile */
218  setValue_ = CANTalon::SetValueMotionProfileEnable;
219  /* MP will start once the control frame gets scheduled */
220  state_ = 2;
221  loopTimeout_ = kNumLoopsTimeout;
222  }
223  break;
224  case 2: /* check the status of the MP */
225  /*
226  * if talon is reporting things are good, keep adding to our
227  * timeout. Really this is so that you can unplug your talon in
228  * the middle of an MP and react to it.
229  */
230  if (status_.isUnderrun == false) {
231  loopTimeout_ = kNumLoopsTimeout;
232  }
233  /*
234  * If we are executing an MP and the MP finished, start loading
235  * another. We will go into hold state so robot servo's
236  * position.
237  */
238  if (status_.activePointValid && status_.activePoint.isLastPoint) {
239  /*
240  * because we set the last point's isLast to true, we will
241  * get here when the MP is done
242  */
243  setValue_ = CANTalon::SetValueMotionProfileHold;
244  state_ = 0;
245  loopTimeout_ = -1;
246 
247  for (int i = 0; i < trajLength_; i++) {
248  free(profile_[i]);
249  }
250  free(profile_);
251 
252  isDone_ = true;
253  }
254  break;
255  }
256  }
257  /* printfs and/or logging */

268  {
269  /* create an empty point */

271

283

287  */
288  talon_.ClearMotionProfileTrajectories();
289 
290  /* This is fast since it's just into our TOP buffer */
291  for(int i=0;i<totalCnt;++i){
292  /* for each point, fill our structure and pass it to API */
293  point.position = profile[i][0]; /*
294  * copy the position, velocity, and

296  */
297  point.velocity = profile[i][1];
298

304  point.velocityOnly = false; /*
305  * set true to not do any position

307  */
308

311  point.zeroPos = true; /* set this to true on the first point */
312

317  */
